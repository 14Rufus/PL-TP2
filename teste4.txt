%%LEX
%literals = "+-/*=()"
%ignore = " \t\n"
%tokens = [ 'VAR','NUMBER' ]
@[a-zA-Z_][a-zA-Z0-9_]*@ return()
@\d+(\.\d+)?@ return()
@.@ error()

%% YACC
%precedence = [
('left','+','-'),
('left','*','/'),
('right','UMINUS'),
]
ts = { }
stat : VAR '=' exp { }
stat : exp {}
exp : exp '+' exp { }
exp : exp '-' exp { }
exp : exp '*' exp { }
exp : exp '/' exp { }
exp : '-' exp %neg UMINUS { }
exp : '(' exp ')' { }
exp : NUMBER { }
exp : VAR { }
